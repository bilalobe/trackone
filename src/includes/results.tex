\subsection{Security Validation}

\textbf{AEAD integrity testing:}
\begin{itemize}[nosep]
    \item Ciphertext modification $\rightarrow$ authentication failure (tag mismatch)
    \item Tag modification $\rightarrow$ verification rejection (immediate detection)
    \item AAD tampering $\rightarrow$ tag mismatch detection (authenticated metadata)
    \item Nonce validation $\rightarrow$ length enforcement (24 bytes for XChaCha20)
    \item Key mismatch $\rightarrow$ decryption fails securely (no partial plaintext leak)
\end{itemize}

\textbf{Replay protection enforcement:}
\begin{itemize}[nosep]
    \item Duplicate frames $\rightarrow$ rejected with logged warnings
    \item Out-of-window jumps $\rightarrow$ rejected (configurable window size, default 64)
    \item Backward sequence numbers $\rightarrow$ rejected (monotonic counter requirement)
    \item Persistence verified: Gateway restart preserves last-seen state per device
    \item Multi-device isolation: Separate replay windows per \texttt{device\_id}
\end{itemize}

\textbf{Cryptographic test coverage:}
\begin{itemize}[nosep]
    \item X25519 ECDH: Known-answer tests, edge cases (zero scalar, base point)
    \item HKDF: RFC 5869 test vectors, empty salt/info handling
    \item XChaCha20-Poly1305: Encryption/decryption roundtrip, tag validation
    \item Ed25519: Signature generation/verification, malformed signature rejection
    \item Random edge cases: All-zeros keys, maximum-length inputs, boundary conditions
\end{itemize}

\subsection{OpenTimestamps Integration}

\textbf{Production Bitcoin anchoring validated:}
\begin{itemize}[nosep]
    \item Day blobs successfully anchored to Bitcoin blocks 919384, 919388, 919418
    \item Multiple calendar server attestations received (a.pool.opentimestamps.org, etc.)
    \item Local verification with Bitcoin Core (headers-only mode, pruned-safe)
    \item Merkle root recomputation matches anchored commitment (end-to-end consistency)
\end{itemize}

\textbf{OTS workflow timing characteristics:}
\begin{itemize}[nosep]
    \item \texttt{ots stamp}: Immediate submission ($<$5 seconds to calendar servers)
    \item \texttt{ots upgrade}: 1--6 hours (awaiting Bitcoin block confirmation)
    \item \texttt{ots verify}: Instant validation (if headers available locally)
    \item Pending proofs: Cannot be verified until Bitcoin block inclusion
\end{itemize}

\textbf{CI/CD integration strategy (ADR-007):}
\begin{itemize}[nosep]
    \item Headers-only Bitcoin Core: $<$1 GB storage (vs. $>$500 GB full node)
    \item Initial header sync: $\sim$15 minutes (one-time cost per CI runner)
    \item Cached headers: Subsequent verifications in seconds
    \item Graceful fallback: Tests use placeholder proofs when OTS unavailable
    \item Real OTS tests: Optional via \texttt{RUN\_REAL\_OTS=1} (requires \texttt{ots} binary)
\end{itemize}
